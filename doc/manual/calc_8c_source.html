100 ERROR_INFO Calc(char *expr, double *result) {
101  queue infix, postfix;
102  ERROR_INFO ret = {NOT_ERROR};
103  if (*expr) {
104  if ((ret = exprAnalyzer(expr, &infix)).error) {
105  return ret;
106  }
107  if ((ret = convert2Postfix(infix, &postfix)).error) {
108  return ret;
109  }
110  if ((ret = calcPostfix(postfix, result)).error) {
111  return ret;
112  }
113  }
114  return ret;

178 static ERROR_FLAG calcSingle(char *op, double num[], double *result) {
179  switch (strlen(op)) {
180  case 1:
181  switch (*op) {
182  case '+': *result = num[0] + num[1]; break;
183  case '-': *result = num[0] - num[1]; break;
184  case '*': *result = num[0] * num[1]; break;
185  case '/':
186  if (num[1]) {
187  *result = num[0] / num[1];
188  } else {
189  // puts("0 can't be used as divide number")
190  return ERROR_DIVIDE_ZERO;
191  }
192  break;
193  case '%':
194  if ((int) num[0] == num[0] && (int) num[1] == num[1]) {
195  *result = (int) num[0] % (int) num[1];
196  } else {
197  // puts("% operation demands integers");
198  return ERROR_MOD_NONINTERGER;
199  }
200  break;
201  case '^': *result = pow(num[0], num[1]); break;
202  case '=': *result = (num[0] == num[1]); break;
203  }
204  break;
205  default:
206  if (!strcmp(op, "sin")) {
207  *result = sin(num[0]);
208  } else if (!strcmp(op, "cos")) {
209  *result = cos(num[0]);
210  } else if (!strcmp(op, "ln")) {
211  if (num[0]<=0){
212  return ERROR_NEGATIVE_IN_LOG;
213  }
214  *result = log(num[0]);
215  } else if (!strcmp(op, "exp")) {
216  *result = exp(num[0]);
217  } else if (!strcmp(op, "pow")) {
218  *result = pow(num[0], num[1]);
219  }
220  break;
221  }
222  return NOT_ERROR;
223 }

267  opNode tempNode = {TYPE_OP, .op = ""};
268  tempNode.op[0] = *(expr++);
269  tempNode.op[1] = '\0';
270  infix.push(&infix, &tempNode);

280  }
281  }
282  *infixRes = infix;
283  return SUCCESS;

286 static const char *numberFetcher(const char *expr, double *result, char *number) {
287  char buf[MAX_NUMBER + 1], *numstr = buf, error = 0;
288  const char *start = expr;
289  if (*expr == '+' || *expr == '-') {
290  *(numstr++) = *(expr++);
291  }
292  if (*expr == '.') {
293  *(numstr++) = '0';
294  }
295  while (isdigit(*expr) || *expr == '.') {
296  if (isdigit(*expr)) {
297  *(numstr++) = *(expr++);
298  } else if (expr != start) { // && *expr == '.'
299  char *temp = numstr - 1, flag = 0;
300  while (temp >= buf) {
301  if (*(temp--) == '.') {
302  flag = 1;
303  break;
304  }
305  }
306  *(numstr++) = *(expr++);
307  if (flag) {
308  error = 1; // one number with two (or more) '.' e.g. 1.23.5
309  }
310  } else {
311  *(numstr++) = *(expr++);
312  }
313  }
314  *numstr = '\0';
315  strcpy(number, buf);
316  sscanf(buf, "%lf", result);
317  return error ? NULL : expr;
318 }

338 static int bracketMatched(const char *expr) {
339  stack brac_stack = newStack(char);
340  char buf;
341  while (*expr != '\0') {
342  switch (*expr) {
343  case '{':
344  case '[':
345  case '(': brac_stack.push(&brac_stack, expr); break;
346  case '}':
347  case ']':
348  if (brac_stack.pop(&brac_stack, &buf) == POP_IN_EMPTY_STACK || buf + 2 != *expr) {
349  brac_stack.destory(&brac_stack);
350  return 0;
351  }
352  break;
353  case ')':
354  if (brac_stack.pop(&brac_stack, &buf) == POP_IN_EMPTY_STACK || buf != '(') {
355  brac_stack.destory(&brac_stack);
356  return 0;
357  }
358  break;
359  }
360  ++expr;
361  }
362  if (brac_stack.empty(&brac_stack)) {
363  brac_stack.destory(&brac_stack);
364  return 1;
365  } else {
366  brac_stack.destory(&brac_stack);
367  return 0;
368  }

371 static ERROR_INFO convert2Postfix(queue infix, queue *postfixRes) {
372  queue postfix = newQueue(opNode);
373  stack opstack = newStack(opNode);
374 
375  while (!infix.empty(&infix)) {
376  opNode tempNode;
377  infix.pop(&infix, &tempNode);
378 
379  if (tempNode.type == TYPE_NUM) {
380  postfix.push(&postfix, &tempNode);
381  } else if (tempNode.type == TYPE_FUNC) {
382  opstack.push(&opstack, &tempNode);
383  } else if (!strcmp(tempNode.op, ",")) {
384  while (!opstack.empty(&opstack)) {
385  opNode optemp;
386  opstack.pop(&opstack, &optemp);
387  if (isLeftBrac(optemp.op[0])) {
388  opstack.push(&opstack, &optemp);
389  break;
390  } else {
391  postfix.push(&postfix, &optemp);
392  }
393  }
394  if (opstack.empty(&opstack)) {
395  // puts("wrong postion for function argument spliter ','");
396  postfix.destory(&postfix); // caution leakage
397  opstack.destory(&opstack);
398  infix.destory(&infix);
399 
400  return (ERROR_INFO){ERROR_WRONG_POSITION_ARGUMENT_SPLITER};
401  }
402  } else if (tempNode.type == TYPE_OP) {
403  while (!opstack.empty(&opstack)) {
404  if ((op_prior(((opNode *) opstack.top(&opstack))->op) > op_prior(tempNode.op)) ||
405  (op_prior(((opNode *) opstack.top(&opstack))->op) == op_prior(tempNode.op) && op_associate(tempNode.op) == LEFT)) {
406 
407  opNode optemp;
408  opstack.pop(&opstack, &optemp);
409  postfix.push(&postfix, &optemp);
410  } else {
411  break;
412  }
413  }
414  opstack.push(&opstack, &tempNode);
415  } else if (isLeftBrac(tempNode.op[0])) {
416  opstack.push(&opstack, &tempNode);
417  } else if (isRightBrac(tempNode.op[0])) {
418  while (!opstack.empty(&opstack) && !isLeftBrac(((opNode *) opstack.top(&opstack))->op[0])) {
419  opNode optemp;
420  opstack.pop(&opstack, &optemp);
421  postfix.push(&postfix, &optemp);
422  }
423  opstack.pop(&opstack, NULL); // clean left bracket
424  // if there is no left bracket, there must be somthing wrong with bracketMatched()
425  }
426  }
427  infix.destory(&infix);
428  while (!opstack.empty(&opstack)) {
429  opNode tempop;
430  opstack.pop(&opstack, &tempop);
431  postfix.push(&postfix, &tempop);
432  }
433  opstack.destory(&opstack);
434 
435  *postfixRes = postfix;
436  return SUCCESS;
437 }

439 static ERROR_INFO calcPostfix(queue postfix, double *result) {
440  stack calcstack = newStack(double);
441  opNode tempNode;
442  while (postfix.pop(&postfix, &tempNode) != POP_IN_EMPTY_QUEUE) {
443  if (tempNode.type == TYPE_NUM) {
444  calcstack.push(&calcstack, &tempNode.num);
445  } else {
446  unsigned int arg_needNum = op_argnum(tempNode.op); // unsigned is neccessary OR you'll get a wierd error with gcc8 or newer
447  double num[arg_needNum]; // C99: VLA
448  for (int i = 1; i <= arg_needNum; ++i) {
449  if (!calcstack.empty(&calcstack)) {
450  calcstack.pop(&calcstack, num + arg_needNum - i);
451  } else {
452  // puts("argument number unmatched");
453  postfix.destory(&postfix);
454  calcstack.destory(&calcstack);
455  return (ERROR_INFO){ERROR_WRONG_OPRAND_NUMBER};
456  }
457  }
458  double temp;
459  ERROR_INFO ret = {NOT_ERROR};
460  if ((ret.error = calcSingle(tempNode.op, num, &temp))) {
461  postfix.destory(&postfix);
462  calcstack.destory(&calcstack);
463  return ret;
464  } else {
465  calcstack.push(&calcstack, &temp);
466  }
467  }
468  }
469  if (calcstack.size == 1) {
470  calcstack.pop(&calcstack, result);
471  } else if (calcstack.size > 1) {
472  // puts("argument number unmatched");
473  postfix.destory(&postfix);
474  calcstack.destory(&calcstack);
475  return (ERROR_INFO){ERROR_WRONG_OPRAND_NUMBER};
476  }
477  postfix.destory(&postfix);
478  calcstack.destory(&calcstack);
479  return SUCCESS;
480 }
